/**
 * Modern function for reading XML configurations
 * Replaces configValue with better error handling and logging
 */
def getConfigValue = { String id ->
    try {
        def configFile = file("${rootProject.projectDir.path}/app/src/main/assets/config/config.json")
        def manualFile = file("${rootProject.projectDir.path}/app/src/main/assets/config/config_manual.json")
        
        if (configFile.exists()) {
            def json = new groovy.json.JsonSlurper().parseText(configFile.text)
            if (json.containsKey(id)) return json[id].toString()
        }
        if (manualFile.exists()) {
            def manualJson = new groovy.json.JsonSlurper().parseText(manualFile.text)
            if (manualJson.containsKey(id)) return manualJson[id].toString()
        }
        
        throw new GradleException("Configuration '" + id + "' not found in config.json files")
    } catch (Exception e) {
        throw new GradleException("Error reading configuration '" + id + "': " + e.message, e)
    }
}

/**
 * Validates optimal settings platform. Fails build early if configuration is incorrect.
 */
def validateOptimalPlatform = { ->
    def isOptimal = getConfigValue('conf_optimal_settings') == 'true'
    if (!isOptimal) return

    def platform = getConfigValue('conf_platform')
    if (platform == null || platform.trim().isEmpty()) {
        def optimalSettingsFile = file("${rootProject.projectDir.path}/app/src/main/assets/optimal_settings.json")
        def availablePlatforms = []
        
        if (optimalSettingsFile.exists()) {
            def jsonSlurper = new groovy.json.JsonSlurper()
            def profiles = jsonSlurper.parseText(optimalSettingsFile.text)
            profiles.each { profile ->
                if (profile.platform_id) {
                    availablePlatforms.add(profile.platform_id)
                }
            }
        }
        
        def platformsMsg = availablePlatforms.isEmpty() ? "" : "\n\nAvailable platforms to choose from:\n- " + availablePlatforms.join("\n- ")
        throw new GradleException("❌ BUILD FAILED: 'conf_optimal_settings' is true, but 'conf_platform' is empty!\n" +
            "When using optimal settings, you must specify the platform ID to avoid generic extension collisions (like .bin or .iso)." +
            platformsMsg)
    }
}

// Run validation during Gradle configuration phase
validateOptimalPlatform()

/**
 * Generates a unique app identifier from conf_name and conf_core.
 * Converts both strings to lowercase, removes special characters,
 * and combines them with an underscore.
 * 
 * Example: "Sonic The Hedgehog" + "picodrive" -> "sonic_the_hedgehog_picodrive"
 */
def generateConfigId = { String name, String core ->
    def cleanName = name.toLowerCase()
        .replaceAll(/[^a-z0-9]+/, '_')
        .replaceAll(/^_+|_+$/, '')
    
    def cleanCore = core.toLowerCase()
        .replaceAll(/[^a-z0-9]+/, '_')
        .replaceAll(/^_+|_+$/, '')
    
    def generatedId = "${cleanName}_${cleanCore}"
    println "generateConfigId('$name', '$core') = '$generatedId'"
    return generatedId
}

task prepareRom {
    dependsOn generateIcons
    doLast {
        def romName = getConfigValue('conf_rom')
        def romDir = file("${rootProject.projectDir}/app/src/main/assets/rom")
        def backupDir = file("${rootProject.projectDir}/roms_backup")
        backupDir.mkdirs()
        romDir.mkdirs()

        // 1. Move any non-active ROMs OUT of assets/rom/ → roms_backup/
        romDir.listFiles()?.each { f ->
            if (f.name != '.gitkeep' && f.name != romName) {
                def dest = new File(backupDir, f.name)
                if (f.renameTo(dest)) {
                    println "📦 ROM moved to backup: ${f.name}"
                }
            }
        }

        // 2. Move the active ROM IN from roms_backup/ → assets/rom/
        def activeRomInAssets = new File(romDir, romName)
        if (!activeRomInAssets.exists()) {
            def backupRom = new File(backupDir, romName)
            if (backupRom.exists()) {
                if (backupRom.renameTo(activeRomInAssets)) {
                    println "🎮 ROM staged: ${romName} (${String.format('%.1f MB', activeRomInAssets.length() / 1048576.0)})"
                }
            } else {
                throw new GradleException("ROM '$romName' not found in assets/rom/ or roms_backup/. Place your ROM file in app/src/main/assets/rom/")
            }
        } else {
            println "🎮 ROM already staged: ${romName} (${String.format('%.1f MB', activeRomInAssets.length() / 1048576.0)})"
        }
    }
}

task prepareCore {
    doLast {
        // Determine core name: either from config.xml or from optimal settings JSON
        def coreName
        def optimalSettingsEnabled = getConfigValue('conf_optimal_settings') == 'true'
        
        if (optimalSettingsEnabled) {
            println "🤖 Optimal Settings enabled - resolving core from platform profile"
            
            def jsonFile = file("${rootProject.projectDir}/app/src/main/assets/optimal_settings.json")
            if (!jsonFile.exists()) {
                throw new GradleException("optimal_settings.json not found in assets/")
            }
            
            def jsonSlurper = new groovy.json.JsonSlurper()
            def profiles = jsonSlurper.parse(jsonFile)
            
            def platformId = getConfigValue('conf_platform')
            def romName = getConfigValue('conf_rom')
            def extension = romName.substring(romName.lastIndexOf('.')).toLowerCase()
            
            println "   Platform ID: ${platformId ?: '(not set)'}"
            println "   ROM: $romName"
            println "   Extension: $extension"
            
            def profile = null
            
            // Try explicit platform ID first
            if (platformId && platformId != '') {
                profile = profiles.find { it.platform_id == platformId }
                if (profile) {
                    println "   ✅ Found profile by platform_id: ${profile.platform_id}"
                }
            }
            
            // Fall back to extension matching
            if (!profile) {
                profile = profiles.find { it.extensions.contains(extension) }
                if (profile) {
                    println "   ✅ Found profile by extension: ${profile.platform_id}"
                } else {
                    throw new GradleException("No optimal settings profile found for extension '$extension'. Add conf_platform or use manual mode (conf_optimal_settings=false)")
                }
            }
            
            coreName = profile.core
            println "   🎯 Selected core: $coreName (from optimal profile)"
        } else {
            coreName = getConfigValue('conf_core')
            println "🛠️  Manual mode - using core from conf_core: $coreName"
        }
        
        def targetAbi = getConfigValue('conf_target_abi')
        def allAbis = ['x86', 'x86_64', 'armeabi-v7a', 'arm64-v8a']
        def abisToFetch = (targetAbi == 'all') ? allAbis : [targetAbi]
        def foundAbis = []
        def missingAbis = []

        println "Target ABI: ${targetAbi == 'all' ? 'all (universal)' : targetAbi}"

        for (abi in abisToFetch) {
            def jniAbiFolder = file("${rootProject.projectDir}/app/src/main/jniLibs/$abi")

            if (jniAbiFolder.exists() && jniAbiFolder.list().length > 0) {
                println "LibRetro core already exists for $abi, skipping download"
                foundAbis << abi
                continue
            }

            def coreUrl = "https://buildbot.libretro.com/nightly/android/latest/$abi/${coreName}_libretro_android.so.zip"

            // Check availability with HEAD request before downloading
            try {
